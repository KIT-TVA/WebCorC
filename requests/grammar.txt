Condition
c ::=      e == e
          | e <= e
          | e >= c
          | e < e
          | e > e
          | c && c
          | c || c
          | c ==> c
          | c <==> c
          | predName(e...e)
          | \forall (t ident); c
          | \exists (t ident); c

e ::= ident | const
            | (e)
            | e + e
            | e - e
            | e / e
            | e * e
            | e % e

t ::= int | [..]
